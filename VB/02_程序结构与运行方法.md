```VBA
End Sub
```
- 文件操作
```VBA
Sub ObjectExample2()
    Dim fso As Object
    Set fso = CreateObject("Scripting.FileSystemObject")
    Dim myFile As Object
    Set myFile = fso.OpenTextFile("C:\workspace\memo.txt", 1) ' 1は読み込みモード
    Debug.Print myFile.ReadAll
    myFile.Close
End Sub

```
